local RunService = game:GetService("RunService")
local Players = game:GetService("Players")
local player = Players.LocalPlayer

-- UI Library
local UILibrary = loadstring(game:HttpGet("https://raw.githubusercontent.com/zxcursedsocute/UI-Library/refs/heads/main/UI-Library.txt"))()
local windows = UILibrary.CreateWindow("MIYU HUB ","","590","V1.1")
local PlayerTab = windows:AddTab("PLAYER","PLAYER")
PlayerTab:AddSection('MAIN')

-- Cấu hình
local COLLECT_RANGE = 25 
local FOLDER_NAME = "DroppedItems"
local OFFSET = Vector3.new(0, 0, 5)
local collector = false 

PlayerTab:AddToggle({
    Name = "Collector Item",
    Description = "Bật để hút - Tắt để đồ rơi tự nhiên",
    Default = false,
    Callback = function(v)
        collector = v
        
        if not v then
            -- LOGIC RESET VỀ TRẠNG THÁI GỐC
            local droppedItems = workspace:FindFirstChild(FOLDER_NAME)
            if droppedItems then
                for _, item in pairs(droppedItems:GetChildren()) do
                    local p = item:IsA("BasePart") and item or item:FindFirstChildWhichIsA("BasePart")
                    if p then
                        p.CanCollide = true -- Bật lại va chạm
                        p.Anchored = false  -- Tắt đóng băng để nó rớt xuống
                        
                        -- Reset vận tốc về 0 rồi cho nó một lực rơi nhẹ
                        p.AssemblyLinearVelocity = Vector3.new(0, -1, 0)
                        p.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
                        
                        -- Đánh thức vật lý: Ép game tính toán lại trọng lực cho món đồ này
                        p:ApplyImpulse(Vector3.new(0, -0.1, 0)) 
                    end
                end
            end
        end
    end,
})

RunService.RenderStepped:Connect(function()
    if not collector then return end
    
    local char = player.Character
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    local droppedItems = workspace:FindFirstChild(FOLDER_NAME)
    if not droppedItems then return end

    local targetPos = (hrp.CFrame * CFrame.new(OFFSET)).Position

    for _, item in pairs(droppedItems:GetChildren()) do
        local p = item:IsA("BasePart") and item or item:FindFirstChildWhichIsA("BasePart")
        if p then
            local distance = (hrp.Position - p.Position).Magnitude
            if distance <= COLLECT_RANGE then
                -- Khi đang hút: Vô hiệu hóa vật lý tạm thời
                p.CanCollide = false
                p.Anchored = false
                p.CFrame = p.CFrame:Lerp(CFrame.new(targetPos), 0.2)
                -- Ép vận tốc bằng 0 để không bị văng khi đang hút
                p.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
            end
        end
    end
end)
